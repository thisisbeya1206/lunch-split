/ DEBT ENGINE (immutable per-order, no netting)  
/
/
/
/
/
/ One debt record per (order × participant).  
/ debtId = orderId + ':' + memberName — stable, deterministic, collision-free.  
/ New orders never invalidate existing payment records.  
/ ══════════════════════════════════════════  
function ordersForMonth(mk){  
return orders.ﬁlter(o => o.date.slice(0,7) === mk);  
}
function calcDebts(mk){  
const result = [];  
ordersForMonth(mk).forEach(o => {  
o.splits.forEach(s => {  
if(s.name === o.payer) return; // payer owes nothing to themselves  
if(s.amount < 0.01) return; // skip zero-amount splits  
result.push({  
debtId: `${o.id}:${s.name}`, // ← stable unique key per debt  
orderId: o.id,  
month: mk,  
title: o.title,  
date: o.date,  
from: s.name,  
to: o.payer,  
// who owes  
// who paid  
amount: +s.amount.toFixed(1)  
);  
}
}
);  
}
);  
/
/ Sort newest ﬁrst, then by amount descending  
return result.sort((a, b) =>  
new Date(b.date) - new Date(a.date) || b.amount - a.amount  
)
;
}
function calcNetBalance(mk){  
const bal = {};  
ordersForMonth(mk).forEach(o => {  
o.splits.forEach(s => {  
if(s.name === o.payer) return;  
if(!bal[o.payer]) bal[o.payer] = 0;  
if(!bal[s.name]) bal[s.name] = 0;  
bal[o.payer] += s.amount;  
bal[s.name] -= s.amount;  
}
);  
}
);  
return bal;  
}

}
function ledgerRemove(id){

function ledgerGet(debtId){  
return payments.ﬁlter(p => p.debtId === debtId);  
}
/
/ Total paid against one debt  
function ledgerPaidTotal(debtId){  
return ledgerGet(debtId).reduce((s, p) => s + p.amount, 0);  
}
/
/ 'unpaid' | 'partial' | 'paid'  
function ledgerStatus(debtId, debtAmount){  
const paid = ledgerPaidTotal(debtId);  
if(paid <= 0.01)  
return 'unpaid';  
if(paid >= debtAmount - 0.01) return 'paid';  
return 'partial';  
}

function migrateLegacyPayments(){  
let dirty = false;  
/
/ ── Step 1: migrate ls3-mpay (v1 snapshot schema) ──  
const legacyMpay = JSON.parse(localStorage.getItem('ls3-mpay') || '{}');  
if(Object.keys(legacyMpay).length){  
Object.entries(legacyMpay).forEach(([month, pairs]) => {  
Object.entries(pairs).forEach(([key, val]) => {  
const arrow = key.indexOf('→');  
if(arrow < 0) return;  
const from = key.slice(0, arrow);  
const to = key.slice(arrow + 1);  
let wasPaid = false;  
let paidAt = today();  
let amount = 0;  
if(typeof val.paid === 'boolean' && val.paid){  
wasPaid = true;  
paidAt = val.paidAt || today();  
}
else if(typeof val.paidAmount === 'number' && val.paidAmount > 0){  
wasPaid = true;  
amount = val.paidAmount;  
paidAt = val.paidAt || today();  
}
if(!wasPaid) return;  
/ Map to matching orders (each order split → its own debtId)  
/
const matchingOrders = orders.ﬁlter(o =>  
o.date.slice(0,7) === month &&  
o.payer === to &&  
o.splits.some(s => s.name === from && s.name !== o.payer)  
)
;
if(matchingOrders.length === 0){  
/ No order found — preserve as synthetic legacy entry  
/
const syntheticDebtId = `legacy:${month}:${from}:${to}`;  
if(!payments.some(p => p.debtId === syntheticDebtId)){  
payments.push({  
id:  
'migrated-' + syntheticDebtId,  
debtId: syntheticDebtId,  
month, from, to,  
amount: amount > 0 ? +amount.toFixed(1) : 1,  
paidAt  
});  
dirty = true;  
}
return;  
}
matchingOrders.forEach(o => {  
const split = o.splits.ﬁnd(s => s.name === from);  
if(!split) return;  
const debtId = `${o.id}:${from}`;  
if(payments.some(p => p.debtId === debtId)) return; // already migrated  
payments.push({  
id:  
'migrated-' + debtId,  
debtId,  
month, from, to,  
amount: +split.amount.toFixed(1),  
paidAt  
}
);  
dirty = true;  
);  
}
});  
}
);  
localStorage.removeItem('ls3-mpay');  
dirty = true;  
}
/
/ ── Step 2: migrate v2 pair-based payments (no debtId ﬁeld) ──  
payments.forEach(p => {  
if(p.debtId) return; // already v3  
const matchingOrders = orders.ﬁlter(o =>  
o.date.slice(0,7) === p.month &&  
o.payer === p.to &&  
o.splits.some(s => s.name === p.from && s.name !== o.payer)  
)
;
if(matchingOrders.length === 1){  
p.debtId = `${matchingOrders[0].id}:${p.from}`;  
}
else if(matchingOrders.length > 1){  
/ Pick the order whose split amount is closest to the payment amount  
/
const best = matchingOrders.reduce((prev, o) => {  
const s = o.splits.ﬁnd(s => s.name === p.from);  
const diﬀ = s ? Math.abs(s.amount - p.amount) : Inﬁnity;  
const ps = prev.splits.ﬁnd(s => s.name === p.from);  
const pdiﬀ = ps ? Math.abs(ps.amount - p.amount) : Inﬁnity;  
return diﬀ < pdiﬀ ? o : prev;  
});  
p.debtId = `${best.id}:${p.from}`;  
else {  
}
}
p.debtId = `legacy:${p.month}:${p.from}:${p.to}`;  
dirty = true;  
);  
}
if(dirty) save();  
}

function toggleDebtPaid(safeId, remaining, debtAmount){  
const debtId = decodeURIComponent(escape(atob(safeId)));  
const status = ledgerStatus(debtId, debtAmount);  
if(status === 'paid'){  
/
/ Already paid → undo: remove all payment records for this debt  
ledgerGet(debtId).map(p => p.id).forEach(id => ledgerRemove(id));  
showToast('已撤銷全部付款紀錄');  
else {  
}
/
/ Not fully paid → add payment record for remaining amount  
const allDebts = calcDebts(settleMonth);  
const d = allDebts.ﬁnd(x => x.debtId === debtId);  
if(!d) return;  
ledgerAdd(debtId, d.month, d.from, d.to, remaining);  
showToast(remaining === debtAmount  
?
:
'已標記付款 ✓'  
`新增付款 $${remaining.toFixed(0)} ✓`);  
}
renderSettleView();  
}
